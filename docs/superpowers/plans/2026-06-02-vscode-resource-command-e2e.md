# VS Code Resource Command Tree E2E Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add VS Code E2E coverage for PR #17698's resource-command tree items after PR #17772's E2E harness lands, without merging or rebasing #17772 directly into #17698.

**Architecture:** Keep #17698's product behavior: `aspire describe --include-disabled-commands` feeds enabled and disabled API-visible resource commands into the tree, enabled command child items are executable from their context menu, disabled command child items render but are not executable, hidden/API-only/unknown-state commands do not render, resources with no visible commands do not show a `Commands` group, and CodeLens ignores disabled commands. After #17772 is merged to `main`, merge updated `main` into `e17698`, resolve the shared extension conflicts by preserving #17772's telemetry/E2E infrastructure and #17698's command-tree behavior, then add fixture commands plus UI-level assertions, including old-CLI compatibility.

**Tech Stack:** TypeScript VS Code extension, ExTester-based VS Code E2E tests, Aspire AppHost C# fixture generated by `extension/scripts/run-e2e.js`, xUnit/MTP for CLI mapper tests.

```ts
class ResourceCommandItem extends vscode.TreeItem {
    constructor(
        public readonly commandName: string,
        public readonly commandJson: ResourceCommandJson,
        public readonly resourceItem: ResourceItem,
        parentId: string
    ) {
        const label = commandJson.displayName ?? commandName;
        super(label, vscode.TreeItemCollapsibleState.None);
        this.id = `${parentId}:command:${commandName}`;
        this.tooltip = commandJson.description ?? undefined;

        const isEnabled = isEnabledCommand(commandJson);

        this.iconPath = getResourceCommandIcon(commandName, isEnabled);
        if (isEnabled) {
            this.contextValue = 'resourceCommand:enabled';
        } else {
            this.description = resourceCommandDisabledDescription;
            this.contextValue = 'resourceCommand:disabled';
        }
    }
}
```

```ts
export function createCliWithoutDisabledCommandCapabilityWrapper(): string {
    const wrapperDirectory = fs.mkdtempSync(path.join(os.tmpdir(), 'aspire-e2e-old-cli-'));
    const realCliPath = getCliPath();

    if (process.platform === 'win32') {
        const wrapperPath = path.join(wrapperDirectory, 'aspire-old-cli.cmd');
        fs.writeFileSync(wrapperPath, `@echo off
if "%1"=="config" if "%2"=="info" if "%3"=="--json" (
  echo {"localSettingsPath":"","globalSettingsPath":"","availableFeatures":[],"localSettingsSchema":{"properties":[]},"globalSettingsSchema":{"properties":[]},"capabilities":[]}
  exit /b 0
)
    for %%A in (%*) do (
      if "%%~A"=="--include-disabled-commands" (
        echo Unrecognized command or argument '--include-disabled-commands' 1>&2
        exit /b 1
      )
    )
    "${realCliPath}" %*
    exit /b %ERRORLEVEL%
    `);
        return wrapperPath;
    }

    const wrapperPath = path.join(wrapperDirectory, 'aspire-old-cli');
    fs.writeFileSync(wrapperPath, `#!/usr/bin/env bash
set -euo pipefail
if [[ "$#" -eq 3 && "$1" == "config" && "$2" == "info" && "$3" == "--json" ]]; then
  printf '%s\n' '{"localSettingsPath":"","globalSettingsPath":"","availableFeatures":[],"localSettingsSchema":{"properties":[]},"globalSettingsSchema":{"properties":[]},"capabilities":[]}'
  exit 0
fi
for arg in "$@"; do
  if [[ "$arg" == "--include-disabled-commands" ]]; then
    printf '%s\n' "Unrecognized command or argument '--include-disabled-commands'" >&2
    exit 1
  fi
done
exec ${JSON.stringify(realCliPath)} "$@"
`);
    fs.chmodSync(wrapperPath, 0o755);

    return wrapperPath;
}
```

```ts
export async function getActiveQuickPickLabels(timeoutMs = 30000): Promise<string[]> {
    const input = await VSBrowser.instance.driver.wait(async () => {
        try {
            return await InputBox.create();
        }
        catch {
            return false;
        }
    }, timeoutMs, 'Timed out waiting for active quick pick to appear.');

    return await VSBrowser.instance.driver.wait(async () => {
        try {
            const picks = await input.getQuickPicks();
            const labels = await Promise.all(picks.map(pick => pick.getLabel()));
            return labels.length > 0 ? labels : false;
        }
        catch {
            return false;
        }
    }, timeoutMs, 'Timed out waiting for active quick pick labels.');
}
```
